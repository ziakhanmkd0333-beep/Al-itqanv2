## 📁 React Router Structure (Code)

```jsx
// App.jsx — Complete Route Map

import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom';

<BrowserRouter>
  <Routes>

    {/* ─── PUBLIC ─── */}
    <Route path="/"                          element={<SplashScreen />} />
    <Route path="/home"                      element={<Homepage />} />
    <Route path="/courses"                   element={<CoursesPage />} />
    <Route path="/courses/:courseSlug"       element={<CourseDetail />} />
    <Route path="/admission"                 element={<AdmissionPage />} />
    <Route path="/about"                     element={<AboutPage />} />
    <Route path="/contact"                   element={<ContactPage />} />

    {/* ─── AUTH ─── */}
    <Route path="/login"                     element={<LoginPage />} />
    <Route path="/forgot-password"           element={<ForgotPassword />} />
    <Route path="/reset-password"            element={<ResetPassword />} />

    {/* ─── ADMIN (Protected) ─── */}
    <Route path="/admin" element={<AdminGuard />}>
      <Route path="dashboard"               element={<AdminDashboard />} />
      <Route path="students"                element={<StudentList />} />
      <Route path="students/add"            element={<AddStudent />} />
      <Route path="students/:id"            element={<StudentDetail />} />
      <Route path="students/:id/edit"       element={<EditStudent />} />
      <Route path="teachers"                element={<TeacherList />} />
      <Route path="teachers/add"            element={<AddTeacher />} />
      <Route path="teachers/:id"            element={<TeacherDetail />} />
      <Route path="teachers/:id/edit"       element={<EditTeacher />} />
      <Route path="courses"                 element={<AdminCourseList />} />
      <Route path="courses/add"             element={<AddCourse />} />
      <Route path="courses/:id"             element={<AdminCourseDetail />} />
      <Route path="courses/:id/edit"        element={<EditCourse />} />
      <Route path="admissions"              element={<AdmissionList />} />
      <Route path="admissions/:id"          element={<AdmissionDetail />} />
      <Route path="payments"                element={<PaymentList />} />
      <Route path="payments/:studentId"     element={<StudentPayments />} />
      <Route path="payments/:id/receipt"    element={<PaymentReceipt />} />
    </Route>

    {/* ─── STUDENT (Protected) ─── */}
    <Route path="/student" element={<StudentGuard />}>
      <Route path="dashboard"               element={<StudentDashboard />} />
      <Route path="courses"                 element={<MyCourses />} />
      <Route path="courses/:courseId"       element={<StudentCourseDetail />} />
      <Route path="attendance"              element={<AttendanceHistory />} />
      <Route path="certificates"            element={<Certificates />} />
      <Route path="certificates/:courseId"  element={<CertificateView />} />
      <Route path="profile"                 element={<StudentProfile />} />
    </Route>

    {/* ─── TEACHER (Protected) ─── */}
    <Route path="/teacher" element={<TeacherGuard />}>
      <Route path="dashboard"               element={<TeacherDashboard />} />
      <Route path="students"                element={<MyStudents />} />
      <Route path="students/:studentId"     element={<StudentView />} />
      <Route path="attendance"              element={<AttendanceManager />} />
      <Route path="attendance/:courseId/:date" element={<SessionAttendance />} />
      <Route path="materials"               element={<SessionMaterials />} />
      <Route path="profile"                 element={<TeacherProfile />} />
    </Route>

    {/* ─── FALLBACKS ─── */}
    <Route path="/unauthorized"             element={<Unauthorized />} />
    <Route path="*"                         element={<NotFound />} />

  </Routes>
</BrowserRouter>
```
